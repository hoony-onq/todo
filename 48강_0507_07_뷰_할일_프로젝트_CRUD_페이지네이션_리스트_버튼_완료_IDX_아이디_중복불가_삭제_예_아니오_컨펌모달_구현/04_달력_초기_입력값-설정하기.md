# 달력 초기 입력값 설정하기
1. new Date()  현재날짜 설정 생성
```js
    // 날짜 데이터 사용 되는 경우
    // 날짜 설정하기
    // 쿠키 사용 설정
    // 타임 세일 카운트 다운 날짜 시간 설정
    // 로그인 세션 설정 
    // 인증 3분 카운트 인증번호 설정
    // :
    // 1. 날짜 설정
    let toDay = new Date();
    console.log( '우리나라 표준시 날짜 확인 toDay' );
    console.log( 'let toDay = new Date()' );
    console.log( toDay );

    console.log( '국제 표준시' );
    console.log( 'toDay.toISOString()' );
    console.log( toDay.toISOString() );


    console.log( 'toDay.toDateString()' );
    console.log( toDay.toDateString() );

    console.log( "toDay.toLocaleDateString('ko-KR')" );
    console.log( toDay.toLocaleDateString('ko-KR') );
    console.log( toDay.toLocaleDateString() );


    console.log( "toDay.toLocaleString('ko-KR')" );
    console.log( toDay.toLocaleString('ko-KR') );
    
    console.log( "toDay.toTimeString()" );
    console.log( toDay.toTimeString() );
                    
    console.log( "toDay.toUTCString()" );
    console.log( toDay.toUTCString() );
    
    // 문자열 변경 .toString() 강제 형변환
    const a = 150;   // Number
    const b = '150'; // String
    const c = '300'; // String


    console.log( 'a' )  
    console.log( a )    //숫자(Number 넘버)

    console.log( 'a.toString()' )  
    console.log( a.toString() )  // 숫자(Number 넘버) => 문자형으로 변환

    console.log( 'String(a)' )
    console.log( String(a) )     // 숫자(Number 넘버) => 문자형으로 변환


    console.log( 'b' )  
    console.log( b )      //문자형(String 스트링)

    console.log( 'b' )  
    console.log( Number(b) )      //문자형(String 스트링) => 강제 형변환 => Number()

    // 연산
    let answer = b + c
    console.log ( answer )

    answer = Number(b) + Number(c)
    console.log ( answer )


    // 스트링 => 배열 => 자르기 slice(시작, 끝)
    const str = 'abcdefg'
    console.log( 'abcdefg' )
    console.log( str.length )
    console.log( `str[0] ${str[0]}` )
    console.log( `str[1] ${str[1]}` )
    console.log( `str[2] ${str[2]}` )
    console.log( `str[3] ${str[3]}` )
    console.log( `str[4] ${str[4]}` )
    console.log( `str[5] ${str[5]}` )
    console.log( `str[6] ${str[6]}` )

    
    // console.log( idx, item )
    console.log( 'map() 사용은 오류발생 ' )  // => [] 배열만 가능
    // str.map((item, idx)=>{                    
    //     console.log( idx, item )
    // })


    // 포 넥스트 포문  for next
    console.log( ' 포 넥스트 포문  for next ' )
    // str.length 문자 길이
    console.log( 'str.length 문자 길이' );
    console.log( str.length );

    for(let i=0; i<str.length; i++){
        console.log( `str[${i}] ${str[i]}` )
    }

    console.log( ' 포 인 포문  for(i in str) ' )
    for(i in str){
        console.log( `str[${i}] ${str[i]}` )
    }




    // abc 자르기
    console.log( `str[0][1][2] ${str[0]}${str[1]}${str[2]}` )
    console.log( `str.slice(0, 3) ${str.slice(0, 3)}` )
    
    console.log( `str[3][4][5][6] ${str[3]} ${str[4]} ${str[5]} ${str[6]}` )
    console.log( `str.slice(3) ${str.slice(3)}` ) // 3 ~ 끝
    
    // efg 자르기
    console.log( `str[4][5][6] ${str[4]}${str[5]}${str[6]}` )
    console.log( `str.slice(-3) ${str.slice(-3)}` )  // 끝에서 3글자 추출
    console.log( `str.slice(-5) ${str.slice(-5)}` )  // 끝에서 5글자 추출

    // const str = 'abcdefg'
    // efg 자르기 삭제 나머지만 추출
    console.log( `str[0][1][2][3] ${str[0]}${str[1]}${str[2]}${str[3]}` )
    console.log( `str.slice(0, -3) ${str.slice(0, -3)}` )  // 시작에서 끝까지 단 끝에서 3글자 삭제(제외)


```
